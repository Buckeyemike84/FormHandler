  361.  361             } // check for braces (array names?)

  364.  364                 // (could be more, like "field[name1][name2]

  367.  367                     if ($i == 0) {
  368.  368                         $names[] = substr($name, 0, $part[1]);
  369.  369                     }
  370.  370 
  371.  371                     $names[] = substr($name, $part[1] + 1, strlen($part[0]) - 2);
  372.  372                 }

  375.  375                 // First look for name1, then go into that array and continue with name2, etc.
  376.  376                 $base = $list;

  380.  380                             $base = $base[$keyName];
  381.  381                         } // no value found!
  382.  382                         else {
  383.  383                             $base = null;
  384.  384                         }
  385.  385                     }
  386.  386                 }
  387.  387 
  388.  388                 $cache[$name] = $base;
  389.  389             } else {
  390.  390                 $cache[$name] = null;
  391.  391             }
  392.  392 
  393.  393             // do we have a value?
  394.  394             if ($cache[$name]) {
  395.  395                 // make sure we filter the input
  396.  396                 if ($this->encodingFilter instanceof InterfaceEncodingFilter) {
  397.  397                     // filter the input
  398.  398                     $cache[$name] = $this->encodingFilter->filter($cache[$name]);
  399.  399                 }
  400.  400             }
  401.  401         }
  402.  402 
  403.  403         return $cache[$name];
  404.  404     }

  484.  484     /**
  485.  485      * Return's true if the form is submitted.
  486.  486      *
  487.  487      * Checks if all the fields are in the correct array ($_GET or $_POST)
  488.  488      * (accept for checkboxes and radio buttons, these are not available if they are unckecked)
  489.  489      *
  490.  490      * Please note that if you have a form with only a checkbox or radio button in it,
  491.  491      * this method will always return true (when the request method is correct).
  492.  492      * This is because when the form is submitted and the checkbox / radio button is not selected,
  493.  493      * there is no way to check if the form is really submitted. In this case you should add
  494.  494      * an hidden field, just to check if the form was submitted or not.
  495.  495      *
  496.  496      * @param string $reason
  497.  497      *            This parameter will be filled with the "reason" why the form is not "submitted"
  498.  498      * @return boolean
  499.  499      */
  500.  500     public function isSubmitted(&$reason = null)
  501.  501     {
  502.  502         $reason = '';
  503.  503         if ($this->submitted === null) {
  504.  504             $this->submitted = true;
  505.  505 
  506.  506             // first check if the request method is the same..
  507.  507             if (empty($_SERVER['REQUEST_METHOD'])) {
  508.  508                 $this->submitted = false;
  509.  509                 $reason = 'No request method is known (should be GET or POST)';
  510.  510             } elseif ($this->getMethod() == Form::METHOD_GET && $_SERVER['REQUEST_METHOD'] != 'GET') {
  511.  511                 $reason = 'Request method is incorrect (should be GET).';
  512.  512                 $this->submitted = false;
  513.  513             } elseif ($this->getMethod() == Form::METHOD_POST && $_SERVER['REQUEST_METHOD'] != 'POST') {
  514.  514                 $reason = 'Request method is incorrect (should be POST).';
  515.  515                 $this->submitted = false;
  516.  516             }
  517.  517 
  518.  518             $list = ($this->getMethod() == Form::METHOD_GET ? $_GET : $_POST);
  519.  519 
  520.  520             // if still "true", check the fields
  521.  521             if ($this->submitted) {
  522.  522                 foreach ($this->fields as $field) {
  523.  523                     if ($field instanceof Field\AbstractFormField &&
  524.  524                         !($field instanceof Field\CheckBox || $field instanceof Field\RadioButton) &&
  525.  525                         $field->getName()
  526.  526                     ) {
  527.  527                         $name = $field->getName();
  528.  528 
  529.  529                         // remove possible brackets
  530.  530                         if (($i = strpos($name, '[')) !== false) {
  531.  531                             $name = substr($name, 0, $i);
  532.  532                         }
  533.  533 
  534.  534                         if ($field instanceof Field\UploadField) {
  535.  535                             if (!$field->isDisabled()) {
  536.  536                                 if (!array_key_exists($name, $_FILES)) {
  537.  537                                     $reason = 'Upload field "' . $name . '" does not exists in $_FILES array.';
  538.  538                                     $this->submitted = false;
  539.  539                                 }
  540.  540                             }
  541.  541                         } // selectfields are not in the "_POST" array when they are "multiple" selectable and
  542.  542                         // nothing is selected.
  543.  543                         elseif ($field instanceof Field\SelectField) {
  544.  544                             if (sizeof($field->getOptions()) == 0 && !array_key_exists($name, $list)) {
  545.  545                                 // form can still be submitted.
  546.  546                                 // An empty select with no options is not included in the POST field
  547.  547                             } // of only 1 option can be selected, it should be in the submitted values!
  548.  548                             elseif (!$field->isMultiple() && !array_key_exists($name, $list)) {
  549.  549                                 if (!$field->isDisabled()) {
  550.  550                                     $reason = 'Selectfield "' . $name . '" does not exists in submited data array.';
  551.  551                                     $this->submitted = false;
  552.  552                                 }
  553.  553                             }
  554.  554                         } elseif (!array_key_exists($name, $list)) {
  555.  555                             if (!$field->isDisabled()) {
  556.  556                                 $reason = 'Field "' . $name . '" does not exists in submited data array.';
  557.  557                                 $this->submitted = false;
  558.  558                             }
  559.  559                         }
  560.  560                     } // submit button? Then the name => value should exists!
  561.  561                     elseif ($field instanceof Field\SubmitButton && $field->getName()) {
  562.  562                         if (!$field->isDisabled()) {
  563.  563                             if (!array_key_exists($field->getName(), $list)) {
  564.  564                                 /** @noinspection PhpUndefinedVariableInspection */
  565.  565                                 $reason = 'Submitbutton "' . $name . '" does not exists in submited data array.';
  566.  566                                 $this->submitted = false;
  567.  567                             }
  568.  568                         }
  569.  569                     } // image button? Then the name_x and name_y values should exists!
  570.  570                     elseif ($field instanceof Field\ImageButton && $field->getName()) {
  571.  571                         if (!$field->isDisabled()) {
  572.  572                             if (!array_key_exists($field->getName() . '_x', $list) ||
  573.  573                                 !array_key_exists($field->getName() . '_y', $list)
  574.  574                             ) {
  575.  575                                 /** @noinspection PhpUndefinedVariableInspection */
  576.  576                                 $reason = 'Imagebutton "' . $name . '" should submit a _x and _y values,' .
  577.  577                                     'but they are not in the data array.';
  578.  578                                 $this->submitted = false;
  579.  579                             }
  580.  580                         }
  581.  581                     }
  582.  582                 }
  583.  583             }
  584.  584         }
  585.  585 
  586.  586         // is this form not submitted and we want to enable csrf protection?
  587.  587         if ($this->isCsrfProtectionEnabled()) {
  588.  588             // if we do not have a csrf-token field yet, add it.
  589.  589             if (!$this->getFieldByName('csrftoken')) {
  590.  590                 // Add a hidden 'csrftoken' field.

  592.  592 
  593.  593                 if (!$this->submitted || !$field->isValid()) {
  594.  594                     $field->setValue(Validator\CsrfValidator::generateToken());
  595.  595                 }
  596.  596             }
  597.  597         }
  598.  598 
  599.  599         return $this->submitted;
  600.  600     }
